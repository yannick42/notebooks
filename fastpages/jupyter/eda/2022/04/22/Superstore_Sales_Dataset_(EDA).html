print(cmap)

# for all states in the shapefile
for state in shpreader.Reader(states_shp).records():

  state_name = state.attributes['name'].rstrip('\x00')
  try:
    value = int(df_2[state_name])
  except KeyError:
    value = 0
  #print(state_name, value, cmap[value])

  if value < min_worth:
    color = 'red'
  elif value < 2*min_worth:
    color = 'darkorange'
  elif value < 3*min_worth:
    color = 'yellow'
  else:
    color = cmap[value]

  # add a colored state
  ax.add_geometries(
    [state.geometry],
    ccrs.PlateCarree(),
    facecolor=color,
    edgecolor='black',
    #styler=colorize_state #func
  )
